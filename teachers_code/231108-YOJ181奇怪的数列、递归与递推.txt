#include <stdio.h>

int main()
{
    int a, b;
    int n;

    scanf("%d%d%d", &a, &b, &n);

    int f[10001]; // 0 1 2 .... 10000об╠Й

    f[1] = 1;
    f[2] = 1;

    int i;
    for(i = 3; i <= n; i++) {
        f[i] = (a * f[i - 1] + b * f[i - 2]) % 7;
    }

    printf("%d\n", f[n]);

    return 0;
}
